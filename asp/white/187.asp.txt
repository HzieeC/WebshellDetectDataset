Sub SEdit()
username=CheckStr(trim(request("username")))
Admin_Pass=CheckStr(trim(request("Admin_Pass")))
if len(username)<2 then
Call Alert ("用户名不能少于2位",-1)
else
set rs=server.CreateObject("ADODB.RECORDSET")
sql="Select top 1 * from ["&tbname&"_Admin]"
rs.open sql,conn,1,3
rs("Admin_Name")=username
If Admin_Pass<>"" then
rs("Admin_Pass")=Mid(md5(Admin_Pass,32),4,18)
end if
rs.update
rs.close
set rs=nothing
Call Alert ("修改成功!","Admin_admin.asp")
end if
End Sub
