		var check_fl = document.saveform.BoardId.value

				document.getElementById("BoardId_tip").innerHTML="<font color='#F60'>分类不能为空</font>";
				flag[1]=0;
				check_data();
			}
			else{
				document.getElementById("BoardId_tip").innerHTML="<font color='green'>√</font>";
				flag[1]=1;
				check_data();
			}
	}
	function checktitle(){
		var check_title = document.saveform.title.value
			if(check_title.length==0){
				document.getElementById("title_tip").innerHTML="<font color='#F60'>标题不能为空</font>";
				flag[2]=0;
				check_data();
			}
			else{
				document.getElementById("title_tip").innerHTML="<font color='green'>√</font>";
				flag[2]=1;
				check_data();
			}
	}
	function checkconten1t(){
			var check_content = editor.getSource();
			if(check_content.length==0){
				document.getElementById("content_tip").innerHTML="<font color='#F60'>内容不能为空</font>";
				flag[3]=0;
				check_data();
			}
			else{
				document.getElementById("content_tip").innerHTML="<font color='green'>√</font>";
				flag[3]=1;
				check_data();
			}
	}
	
	function regcheck(formct){
		if (editor.getSource()=='' ){

			document.getElementById("content_tip").innerHTML="<font color='#F60'>内容不能为空</font>";

			flag[3]=0;
			check_data();
		}
		else{
			document.getElementById("content_tip").innerHTML="<font color='green'>√</font>";
			flag[3]=1;
			check_data();
		}
	}

	function check_data(){
		if(flag[1]==1 && flag[2]==1 && flag[3]==1){
