function collapse(img, objName)
{
	var obj;
	obj = document.getElementById(objName);
	if (img.src.indexOf('open') != -1)
	{
		img.src = img.src.replace('open', 'close');
		obj.style.display = 'none';
	}
	else
	{
		img.src = img.src.replace('close', 'open');
		obj.style.display = '';
	}
}
