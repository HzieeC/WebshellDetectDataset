<% '添加数据到数据表
act=Request("act")

l_name=trim(request.form("name"))

'检测文件夹名是否与系统文件夹名重复

else

rs("FileName")=l_url

rs("image")=l_image

rs.update
rs.close
set rs=nothing
%>
