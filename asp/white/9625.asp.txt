function CheckIfEnglish(String)
			{ 
			    var Letters = "1234567890.";
			     var i;
			     var c;
			     for( i = 0; i < String.length; i ++ )
			     {
			          c = String.charAt( i );
				  if (Letters.indexOf( c ) < 0)
				     return false;
			     }
			     return true;
			}
