  While Not rs.BOF	' 判断是否过了最后一项
%>

    <title><%=rs(1)%></title>
    <link><%=rs(0)%></link>
    <pubDate><%=rs(2)%></pubDate>
    <data>share</data>
</item>

<%
 rs.MovePrevious	' 移到下一项
Wend

set rs=nothing
%>
