function getnewsclass(cid)
dim rsc
set rsc=createobject("adodb.recordset")
rsc.open "select * from news_class where nid="&cid,conn,1,1
if not rsc.eof then
getnewsclass=rsc("title")
else
getnewsclass="未分类"
end if
rsc.close
set rsc=nothing

end function
