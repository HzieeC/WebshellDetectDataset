function encodestr(str)
str=trim(str)
str=replace(str,"<","&lt;")
str=replace(str,">","&gt;")
str=replace(str,"'","""")
str=replace(str,vbCrLf&vbCrlf,"</p><p>")
str=replace(str,vbCrLf,"<br>")
encodestr=replace(str,"  ","　")
end function
