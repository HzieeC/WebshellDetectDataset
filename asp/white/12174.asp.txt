'备份数据库管理
Private Sub BackupManage()
	Call ShareTableTop2("share_list.gif","","备份数据库管理")
	Call ShareTableItemTitle("6%,54%,12%,18%,10%","编号,备份名称,文件大小,备份时间,下载  删除")

	Set showrec=server.CreateObject("ADODB.recordset")
	showrec.open "select * from OT_collBackupDB order by CB_ID DESC",collobj,3,1
		If showrec.EOF Then
			response.write(_
			"<tr><td class='font1_1' style='text-align:center; padding:8px;' colspan='20'>暂无备份数据库</td></tr>")
		Else
			showPageSize=memberexe("MB_itemNum")		'每页条数
			showrec.pagesize=showPageSize	
			showPageCount=showrec.pagecount
			showCount=showrec.recordcount
			page=ToGetInt("page")
			if page < 1 Or page > showPageCount Then page=1
			showrec.absolutepage=page
			
			response.write(_
			"<tbody class='tabBody padd5'>"&_
			"")
			number=1+(page-1)*showPageSize
			For i=1 To showPageSize
				If showrec.EOF Then Exit For
				If i Mod 2 = 0 Then bgcolor="class='tabColorTr'" Else bgcolor=""
			response.write(_
			"<tr id='closeData"& showrec("CB_ID") &"' "& bgcolor &">"&_
				"<td align='center'>"& number &"</td>"&_
				"<td>"& showrec("CB_filePath") &"</td>"&_
				"<td align='center'>"& FileSizeUnit(showrec("CB_fileSize")) &"</td>"&_
				"<td align='center'>"& showrec("CB_time") &"</td>"&_
				"<td align='center'>"&_
					"<img src='images/img_down.gif' style='cursor:pointer' onclick=""var a=window.open('collDatabase_deal.asp?mudi=download&amp;nohrefStr=close&amp;dataID="& showrec("CB_ID") &"')"" alt='' />"&_
					"&nbsp;&nbsp;"&_
					"<img src='images/img_del.gif' style='cursor:pointer' onclick=""if(confirm('你确定要删除？')==true){DataDeal.location.href='collDatabase_deal.asp?mudi=backupDel&amp;dataID="& showrec("CB_ID") &"'}"" alt='' />"&_
				"</td>"&_
			"</tr>"&_
			"")
			showrec.movenext
			number=number+1
			Next
			response.write(_
			"</tbody>"&_
			"")
		End If
	showrec.close
	Set showrec=Nothing

	Call ShareTableBottom2()
End Sub
