Function c2u(sGBStr)
Dim i,c
if sGBStr<> "" then
For i = 1 to Len(sGBStr)
c = Mid(sGBStr,i,1)

c2u = c2u & "<br/>"
else
c2u = c2u & "&#x" & Hex(AscW(c)) & ";"
end if
Next

End Function
