<%
Function adminlogin(ft)
	AdminName=session("admin_user")
    If AdminName<>"" Then
		sql="select top 1 * from "&ft&"ftbbs_admin Where admin_user='"&AdminName&"'"
        Set rs = Server.CreateObject("ADODB.Recordset")
        rs.open sql,conn,1,1
        if rs.bof and rs.eof then
            adminlogin="N"
        else
            If trim(rs("admin_pwd"))=trim(session("admin_pwd")) Then
               adminlogin="Y"
            Else
               adminlogin="N"
            End If
        end IF
        rs.close
        set rs=nothing
    else
        adminlogin="N"
    end if    
End Function
if adminlogin(ft)="N" then
   response.write "<br><br><br><br><center>非法登录或登录超时!</center>"
   response.End()
end if
%>
