	Function GetModeID(ub,num,sttemp)
		Dim tempstr,i
 		For i=0 To ub	
			If i=num Then 
				tempstr=tempstr&newm(sttemp(i))&"§"
			Else 
				If i=ub Then 
					tempstr=tempstr&sttemp(i)
				Else 
					'echo sttemp(0)
					tempstr=tempstr&sttemp(i)&"§"
				End If 
			End If 

		Next 
		GetModeID=tempstr
	End Function
