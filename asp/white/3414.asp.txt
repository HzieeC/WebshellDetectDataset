function formsubmit(frm,action)
{
	if(frm.SN.value.trim()=="")
	{
		ShowErrMsg("广告编号不能为空，请输入");
		frm.SN.focus();
		return false;
	}
	if(frm.Title.value.trim()=="")
	{
		ShowErrMsg("广告描述不能为空，请输入");
		frm.Title.focus();
		return false;
	}
	if(frm.Code.value.trim()=="")
	{
		ShowErrMsg("广告代码不能为空，请输入");
		frm.Code.focus();
		return false;
	}
	
	frm.action.value = action;
	frm.submit();
}
