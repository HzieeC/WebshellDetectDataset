Function Replace_Mess(fString)
if isnull(fString) then
Replace_Mess=":T02:"
exit function

fString=trim(fString)
fString=replace(fString,"","―")

fString=replace(fString,"　","―")
fString=replace(fString,"'","“")
fString=replace(fString,"--","―")

fString=replace(fString,"update","―")
