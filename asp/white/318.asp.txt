<% '添加数据到数据表

rs("memo")=request("memo")
rs("time")=now()

else
rs.addnew

rs("memo")=request("memo")
rs("time")=now()
end if
rs.update
rs.close
set rs=nothing

fso.copyfile filesource,fileto,true

end if
set fso = nothing

response.Write "<script language='javascript'>alert('备份成功！');location.href='Data_list.asp';</script>"

end if
end if
