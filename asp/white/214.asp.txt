TrueName = 	CheckStr(trim(request.form("TrueName")))

birthday = 	CheckStr(request.form("birthday"))
qm = 		trim(request.form("qm"))

	If Not Checkpost(True) Then Call Alert("禁止外部提交!","-1")
	set rs = server.CreateObject ("adodb.recordset")
	sql="select * from "&tbname&"_user where UserName='"&UserName&"'"
	rs.open sql,conn,1,3
		if Email="" or City="" then
			Call Alert("请填写完整再提交","-1")
		end if
		If PassWord1<>PassWord2 then
			Call Alert("两次输入的密码不同","-1")
		End if
		
		If PassWord1<>"" then
		rs("PassWord")			=md5(PassWord1,16)
		end if
		rs("Sex")				=Sex
		rs("Email")				=Email
		rs("UserQQ")			=QQ
		rs("Province")			=Province
		rs("City")				=City
		If birthday<>"" then
		rs("birthday")			=birthday
		end if
		rs("qm")				=left(qm,200)
		rs.update
		Response.Write("<script language=javascript>alert('恭喜你,修改成功');this.top.location.href='?action=useredit';</script>")
		rs.close
		Set rs = nothing
end sub
