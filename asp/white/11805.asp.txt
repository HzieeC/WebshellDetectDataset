	function BoxDel()
	{
		var bid = document.getElementsByName("BoxId");
		var chkflag = true;
		var delboxid = "0";
		for(var i = 0; i < bid.length; i++)
		{
			if(bid[i].checked == true)
			{
				chkflag = false;
				break;
			}
		}
		if(chkflag)
		{
			alert("请先选择要删除的企业");
			return false;
		}
		var question = confirm("你确定要删除选中的企业吗？");
		if (question !="0")
		{
			for(i = 0; i < bid.length; i++)
			{
				if(bid[i].checked == true)
				{
					delboxid += "|" + bid[i].value;
				}
			}
			//alert(delboxid);
			window.showModalDialog("DelComPool.asp?PoolId=" + delboxid,"DelComPool","");
			location.reload();
		}
	}

	function selAll()
	{
		var selobj = document.getElementsByName("BoxId");
		for(var i = 0; i < selobj.length; i++)
		{
			selobj[i].checked = true;
		}
	}
	
	function removeAll()
	{
		var selobj = document.getElementsByName("BoxId");
		for(var i = 0; i < selobj.length; i++)
		{
			selobj[i].checked = false;
		}
	}
	
	function onclicksel()
	{
		var chkobj = document.all("BoxIdAll");
		if(chkobj.checked == true)
		{
			selAll();
		}
		else
		{
			removeAll();
		}
	}
