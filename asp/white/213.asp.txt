function thehead 
headrs=server.CreateObject("adodb.recordset")

response.Write("<li>上一篇：没有了</li>") 
else

If html=1 then
response.Write("<li>上一篇：<a href='../List.asp?ID="&a0&"'>"&a1&"</a></li>")

end function
