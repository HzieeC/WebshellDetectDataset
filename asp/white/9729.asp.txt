 function upload(id,ip) 
{
  ( new J.dialog({ id:'ip'+id ,title:'查看IP', loadingText:'网页加载中...',  link:true,page: 'http://www.fycms.com/ip/?q='+ip+ "&" + Math.random(), width:700, height:240 })).ShowDialog();

function show(id)
{
	if(document.all(id).style.display=='none')
{
	document.all(id).style.display='block';
}
else
{
	document.all(id).style.display='none';
}
}
function overColor(Obj)
{
	var elements=Obj.childNodes;
	for(var i=0;i<elements.length;i++)
	{
		elements[i].className="tdbg1"
		Obj.bgColor="";
	}
	
}
function outColor(Obj)
{
	var elements=Obj.childNodes;
	for(var i=0;i<elements.length;i++)
	{
		elements[i].className="tdbg";
		Obj.bgColor="";
	}
}
 
function delpost(){
    document.SysLog.method="post";
    document.SysLog.action="?Action=Del";

	this.document.SysLog.submit();
	return true;}return false;
}
	}
