sub sadd()

	rs("ex_url")	=	Request.Form("ex_url")
	rs("ex_pl")				=	Request.Form("ex_pl")
	rs("ex_check")		=	Request.Form("ex_check")
rs.Update
rs.Close
Set rs = Nothing
end sub

sub pl()
dim id,nid,eid,ex_pl,nex_pl,ex_check,nex_check
id				=Trim(Request.Form("id"))
nid				=split(id,",")
ex_pl			=request.form("ex_pl")
nex_pl		=split(ex_pl,",")
ex_check	=request.form("ex_check")
nex_check	=split(ex_check,",")
eid				=ubound(nid)
for i=0 to eid
	set rs=server.createobject("adodb.recordset")
	sql="select ex_pl,ex_check from "&web_dbtop&"custom where id="&nid(i)
	rs.open sql,conn,1,3
	rs("ex_pl")=nex_pl(i)
	rs("ex_check")=nex_check(i)
	rs.update
	rs.close
	set rs=nothing
next
end sub

sub del()
	conn.execute "delete from "&web_dbtop&"custom where id="&Trim(Request.QueryString("id"))
end sub
