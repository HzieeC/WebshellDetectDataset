sub savenew()
	Title			=LoseHtml(trim(request.form("Title")))
	LinkUrl			=LoseHtml(trim(request.form("LinkUrl")))
	LogoUrl			=LoseHtml(trim(request.form("LogoUrl")))
	QQ				=LoseHtml(trim(request.form("qq")))
	
	if Title="" or LinkUrl="" or QQ="" then
		Call Alert ("请填写完整","-1")
	end if
	
	If session("linkposttime")<>"" then
		If DateDiff("s",session("linkposttime"),Now())<36 then
		Call Alert ("你提交的太快了,稍后再来申请","-1")
  		end if
	end if
	
	If Not Checkpost(True) Then Call Alert("禁止外部提交!","-1")
	set rs = server.CreateObject ("adodb.recordset")
	sql="select * from "&tbname&"_Link"
	rs.open sql,conn,1,3
		rs.AddNew 
		rs("Title")				=Title
		rs("ClassID")			=1
		rs("LinkUrl")			=LinkUrl	
		rs("LogoUrl")			=LogoUrl
		rs("AddTime")			=Now()
		rs("LastTime")			=Now()+30
		rs("yn")				=0
		rs("Moneys")			=0
		rs("QQ")				=QQ
		rs("Num")				=Mydb("Select Max([Num]) From ["&tbname&"_Link]",1)(0)+1
		rs("IP")				= GetIP
		rs.update
		Session("linkposttime") =Now()
		Response.Write("<div style=""padding:50px 0;text-align:center;color:#ff0000;"">添加成功，等待审核！请在贵站加上本站链接。</div>")
	rs.close
	set rs=nothing
end sub
