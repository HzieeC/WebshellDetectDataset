			var ext = a.split(".")[1];

			{
				var oEditor = FCKeditorAPI.GetInstance('oContent');
				oEditor.InsertHtml( a );
			}else{
			var oEditor = FCKeditorAPI.GetInstance('oContent');
			oEditor.InsertHtml('<img src="' + a + '" />');

			}
		}else
		{
			var err = a.split("|");
			document.getElementById('error').innerHTML=err[1];
		}
}
