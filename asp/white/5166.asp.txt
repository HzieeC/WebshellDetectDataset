<%
newsid = Request.QueryString("id")
Set rs = Server.CreateObject("Adodb.RecordSet")

Else
	rs.Delete
	rs.UpDate
	Response.Write "<script language=javascript>alert('删除成功！');window.history.back();</script>"
	rs.Close
	Set rs = Nothing
End If
