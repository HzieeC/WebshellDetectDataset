   Call Alert ("投票选项"&i+1&"有错误!",-1)
   end if
   next  
   vote=vote&s&"|"
   result=result&s1&"|"
next

	set rs = server.CreateObject ("adodb.recordset")
	sql="Select * from "&tbname&"_Vote where Title='"& Title &"'"
	rs.open sql,conn,1,3

		rs.AddNew
