function orderby(ary,ary1)
	Sortby=""
	KeepChecking = TRUE
	Do until KeepChecking = FALSE
		KeepChecking = FALSE
		For I = 0 to UBound(ary)
			If I = UBound(ary) Then Exit For
			If cint(ary(I)) > cint(ary(I+1)) Then
				FirstValue = ary(I)
				SecondValue = ary(I+1)
				ary(I) = SecondValue
				ary(I+1) = FirstValue
				
				FirstValue1=ary1(I)
				ary1(I)=ary1(I+1)
				ary1(I+1)=FirstValue1
				KeepChecking = TRUE
			End If
		Next
	Loop
	SortBy=ary1
End function
