    partner=""     '支付宝合作id

'*********************获取支付宝POST过来通知消息************************
For Each varItem in Request.Form
	mystr=varItem&"="&Request.Form(varItem)&"^"&mystr
Next 
If mystr<>"" Then 
	mystr=Left(mystr,Len(mystr)-1)
End If 
mystr = SPLIT(mystr, "^")
Count=ubound(mystr)
'对参数排序
For i = Count TO 0 Step -1
	minmax = mystr( 0 )
	minmaxSlot = 0
	For j = 1 To i
		mark = (mystr( j ) > minmax)
		If mark Then 
			minmax = mystr( j )
			minmaxSlot = j
		End If 
	Next
	If minmaxSlot <> i Then 
		temp = mystr( minmaxSlot )
		mystr( minmaxSlot ) = mystr( i )
		mystr( i ) = temp
	End If
Next
'构造md5摘要字符串
For j = 0 To Count Step 1
	value = SPLIT(mystr( j ), "=")
	If  value(1)<>"" And value(0)<>"sign" And value(0)<>"sign_type"  Then
		If j=Count Then
			md5str= md5str&mystr( j )
		Else 
			md5str= md5str&mystr( j )&"&"
		End If 
	End If 
Next
md5str=md5str&key
mysign=md5(md5str)
'**********************************************************************
 
 
 '*******************下面可以填写对应的数据库操作**********************
If mysign=request.Form("sign") And ResponseTxt="true" Then 	
	If request.Form("trade_status") = "WAIT_BUYER_PAY" Then
		'等待买家付款
		returnTxt	= "success"	
	ElseIf trade_status = "WAIT_SELLER_SEND_GOODS" Then      
		'买家付款成功,等待卖家发货
		returnTxt	= "success"		
	ElseIf trade_status = "WAIT_BUYER_CONFIRM_GOODS" Then    
		'卖家已发货等待买家确认
		returnTxt	= "success"	
	ElseIf trade_status = "TRADE_FINISHED" Then             
		'交易成功结束
		returnTxt	= "success"		
	Else                                                     
		'其他交易状态通知情况
		returnTxt	= "success"
	End If
	Response.Write returnTxt
Else
	response.write "fail"
End If

Function DelStr(Str)
	If IsNull(Str) Or IsEmpty(Str) Then
		Str	= ""
	End If
	DelStr	= Replace(Str,";","")
	DelStr	= Replace(DelStr,"'","")
	DelStr	= Replace(DelStr,"&","")
	DelStr	= Replace(DelStr," ","")
	DelStr	= Replace(DelStr,"　","")
	DelStr	= Replace(DelStr,"%20","")
	DelStr	= Replace(DelStr,"--","")
	DelStr	= Replace(DelStr,"==","")
	DelStr	= Replace(DelStr,"<","")
	DelStr	= Replace(DelStr,">","")
	DelStr	= Replace(DelStr,"%","")
End Function
