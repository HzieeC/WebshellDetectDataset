Function RndString(length)
 Dim chars
 chars = "123456789"

 For i = 1 to length
   randomize
   RndString = RndString & Mid(chars, Cint(Rnd()*(Len(chars)-1)) + 1, 1)
 Next

End Function

Function mAsk(Source)
	Source = Replace(Source,"\","\\")
	Source = Replace(Source,"""","\""")
	Source = Replace(Source,"'","\’")
	Source = Replace(Source,",","\，")
	Source = Replace(Source,"(","\（")
	Source = Replace(Source,")","\）")
	Source = Replace(Source,"<","&lt;")
	Source = Replace(Source,">","&gt;")
	Source = Replace(Source,"&amp;nbsp;","")
	Source = Replace(Source," ","")
	Source = Replace(Source,"　","")
	Source = Replace(Source,"&lt;br&gt;","")
	Source = Replace(Source,"&amp;quot;","""")
	Source = Replace(Source,"&#13;","")
	Source = Replace(Source,"&#10;","")
	Source = Replace(Source,chr(10),"")
	Source = Replace(Source,chr(13),"")
	mAsk = Source
End Function

  function cutStr(str,strlen)
 '截取字符串
	dim l,t,c
	l=len(str)
	t=0
	for i=1 to l
	c=Abs(Asc(Mid(str,i,1)))
	if c>255 then
	t=t+2
	else
	t=t+1
	end if
	if t>=strlen then
	cutStr=left(str,i)&"..."
	exit for
	else
	cutStr=str
	end if
	next
	cutStr=replace(cutStr,chr(10),"")
end function

   function getday(str)
   '得到日期
   dim m
   if trim(str)="" or isnull(str) then
   m="17/11"
   else
   m=day(str)&"/"&month(str)
   end if
   getday=m
   end function
