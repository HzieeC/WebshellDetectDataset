function check(field){
	if(checkflag=="false"){
		for(i=0;i<field.length;i++){
			field[i].checked=true;
		}
		checkflag="true";
		return "解除全选";
	}
	else {
		for(i=0;i<field.length;i++) {
			field[i].checked=false;
		}
		checkflag="false";
		return "选择全部";
	}
}
