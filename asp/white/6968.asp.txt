	Private Function Getname()
		Dim y, m, d, h, mm, S, r
		Randomize
		y = Year(Now)
		m = right("0" & Month(Now),2)
		d = right("0" & Day(Now),2)
		h = right("0" & Hour(Now),2)
		mm =right("0" & Minute(Now),2)
		S = right("0" & Second(Now),2)
		r = 0
		r = CInt(Rnd() * 10000)
		S = right("0000" & r,4)
		Getname = y & m & d & h & mm & S & r
	End Function
