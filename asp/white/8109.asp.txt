		for i=0 to ubound(arr_id)
			sqlstr="delete from " & dbs & " where " & editField & "=" & trim(arr_id(i))
			conn.execute(sqlstr)
		next
	else
		sqlstr="delete from " & dbs & " where " & editField & "=" & editid
		conn.execute(sqlstr)
	end if
	call closeconn()
	url_ret="../table_list.asp?dbs=" & dbs

		url_ret=url_ret & "&path=" & path
	end if
